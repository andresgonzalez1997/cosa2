AOE2 (la tuya, para referencia) =

-- Mismo AOE pero evitando duplicados por las llaves del NIS
AOE2 sin duplicar =
VAR T =
    SUMMARIZE (
        'nis',
        'nis'[ohldoc],
        'nis'[Customer Number],
        'nis'[SKU Code],
        'nis'[month_year_nat],
        "v", MAX ( 'nis'[AOE-TaxBase] )
    )
RETURN
SUMX ( T, [v] )
