% Param MM Acumulado :=
VAR _totalFinal =
    CALCULATE (                         -- quita filtros de fila, pero
        MAX ( [Parameter MM Accumulated] ), -- ① toma el mayor acumulado (el último)
        ALLSELECTED ( 'DataBase_SalesData' ) -- ② respeta slicers y filtros externos
    )
RETURN
    DIVIDE ( [Parameter MM Accumulated], _totalFinal )
