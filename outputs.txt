Sku_final :=
VAR _sku = nis[Sku_Code]
VAR _nuevo =
    CALCULATE (
        MAXX (
            FILTER ( 'tb_equivalences_sku',
                'tb_equivalences_sku'[full_code] = _sku
            ),
            'tb_equivalences_sku'[full_code_new]
        )
    )
RETURN
COALESCE ( _nuevo, _sku )
