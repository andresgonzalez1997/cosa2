def _standardize(tbl: pd.DataFrame) -> Optional[pd.DataFrame]:
    """
    Recorta a 16 columnas y corrige desplazamientos (17 columnas),
    **sin perder la columna-categoría** para que luego
    `add_species_column()` la propague.
    """
    # Demasiado corta → descartar.
    if tbl.shape[1] < 16:
        return None

    if tbl.shape[1] >= 17:
        # ── A ── Detectamos, fila a fila, si la 1.ª columna es categoría
        header_col = tbl.iloc[:, 0]
        second_col = tbl.iloc[:, 1]

        # Fila de producto si la segunda columna empieza con dígito
        product_mask = second_col.astype(str).str.strip().str[0].str.isdigit()

        # Copiamos tabla “core” (16 cols) tomando 1:17 como siempre
        core = tbl.iloc[:, 1:17].copy()

        # En filas de CATEGORÍA movemos el texto de la 1.ª columna
        # a la que luego será «product_desc» (índice 2 después del recorte)
        core.loc[~product_mask, core.columns[2]] = (
            header_col.loc[~product_mask]
            .astype(str)
            .str.replace(r",", "", regex=True)  # limpia comas
            .str.strip()
        )
        tbl = core
    else:
        # Caso normal de 16 columnas
        tbl = tbl.iloc[:, :16]

    tbl.columns = COLUMN_NAMES
    return tbl
