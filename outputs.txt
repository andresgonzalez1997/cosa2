Sku_final :=
VAR _nuevoPorCodigo =
    CALCULATE (
        MAXX (
            TOPN (
                1,
                FILTER ( 'tb_equivalences_sku',
                    'tb_equivalences_sku'[full_code] = nis[Sku_Code]
                ),
                'tb_equivalences_sku'[load_date], DESC
            ),
            'tb_equivalences_sku'[full_code_new]
        )
    )
RETURN
COALESCE ( _nuevoPorCodigo, nis[Sku_Code] )
